\begin{frame}{Motivação e Objetivo}
  \begin{columns}
    \begin{column}{0.55\textwidth}
      \begin{itemize}
        \item \textbf{Problema:} Agrupar automaticamente notícias em categorias temáticas \emph{sem supervisão humana} — tarefa clássica e desafiadora em PLN.

        \item \textbf{Questão de Pesquisa:} Qual representação vetorial e qual algoritmo produz grupos mais coerentes com a semântica humana?
      \end{itemize}
    \end{column}
    \begin{column}{0.43\textwidth}
      \begin{block}{Abordagem do Experimento}
        \small
        \begin{enumerate}
          \item Vetorização: TF-IDF vs Embeddings Densos
          \item Clustering: 4 algoritmos distintos
          \item Projeção 2D: PCA, t-SNE, UMAP
          \item Explicação Semântica via LLM Local + Nuvem
        \end{enumerate}
      \end{block}
    \end{column}
  \end{columns}

\section{Representações Vetoriais}
% =============================================
% Slide 4: Representações Vetoriais
% =============================================
\begin{frame}{Representações Vetoriais de Texto}
  \begin{columns}
    \begin{column}{0.5\textwidth}
      \textbf{\color{ufrnblue}TF-IDF (Frequencial)}
      \begin{itemize}
        \item Abordagem clássica: sacola de palavras (bag-of-words)
        \item Apenas frequência léxica — ignora contexto semântico
        \item Vocabulário: até \textbf{1000 termos} após remoção de stopwords PT
        \item Equação:
        \small
        \[ \text{TF-IDF}(t, d, D) = \text{tf}(t,d) \cdot \log\!\left(\frac{|D|}{1 + |\{d:t\in d\}|}\right) \]
      \end{itemize}
    \end{column}
    \begin{column}{0.5\textwidth}
      \textbf{\color{ufrnblue}Embeddings Semânticos Densos}
      \begin{itemize}
        \item Modelos neurais pré-treinados que mapeiam textos em espaços contínuos
        \item Capturam sinônimos, contexto e relações implícitas
        \item \textbf{SentenceTransformers:}\\
          \texttt{all-MiniLM-L6-v2} — 384 dimensões
        \item \textbf{Ollama (local):}\\
          \texttt{nomic-embed-text} — 768 dimensões
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}

\section{Algoritmos de Clustering}
% =============================================
% Slide 5: Algoritmos
% =============================================
\begin{frame}{Algoritmos de Agrupamento Testados}
  \begin{columns}
    \begin{column}{0.5\textwidth}
      \begin{block}{K-Means}
        Agrupamento particional que minimiza a inércia intra-cluster. Assume agrupamentos convexos e de tamanho similar.
      \end{block}
      \vspace{0.3em}
      \begin{block}{Agglomerative (Ward)}
        Hierárquico ascendente (\textit{bottom-up}). Mescla pares de clusters pelo critério de Ward (menor variância dentro do grupo).
      \end{block}
    \end{column}
    \begin{column}{0.5\textwidth}
      \begin{block}{DBSCAN}
        Baseado em densidade. Identifica \textit{outliers} como ruído ($-1$). Sensível ao parâmetro $\epsilon$ e dimensionalidade do espaço.
      \end{block}
      \vspace{0.3em}
      \begin{block}{HDBSCAN}
        Versão hierárquica do DBSCAN. Suporta densidades variáveis sem parâmetro $\epsilon$ fixo. Robusto a outliers.
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\section{Métricas de Avaliação}
% =============================================
% Slide 6: Métricas
% =============================================
\begin{frame}{Métricas de Validação}
  \begin{columns}
    \begin{column}{0.5\textwidth}
      \textbf{\color{ufrnblue}Não Supervisionadas (sem ground truth)}
      \begin{itemize}
        \item \textbf{Coeficiente de Silhueta:}
        \[ s(i) = \frac{b(i) - a(i)}{\max(a(i),b(i))} \in [-1,+1] \]
        onde $a(i)$ = distância intra-cluster, $b(i)$ = distância ao cluster vizinho mais próximo.
        \item \textbf{Davies-Bouldin:} Menor $\Rightarrow$ clusters mais compactos e separados.
      \end{itemize}
    \end{column}
    \begin{column}{0.5\textwidth}
      \textbf{\color{ufrnblue}Supervisionadas (vs. Classes Reais)}
      \begin{itemize}
        \item \textbf{ARI (Adjusted Rand Index):} Concordância corrigida pelo acaso entre partição predita e \textit{ground truth}. Ideal $= 1.0$.
        \item \textbf{NMI (Normalized Mutual Info):} Baseada em teoria da informação.
        \item \textbf{Pureza:} Fração das amostras na classe majoritária de cada cluster.
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}

% =============================================
% Slide 8: Análise do Resultado Surpreendente
% =============================================
\begin{frame}{Por que o TF-IDF Superou os Embeddings?}
  \begin{columns}
    \begin{column}{0.55\textwidth}
      \begin{itemize}
        \item \textbf{Dataset Pequeno (315 amostras):} TF-IDF em corpus pequeno captura bem os termos discriminativos das 6 classes — já que cada classe usa vocabulário específico e consistente.
        \item \textbf{Linguagem Estruturada:} Notícias jornalísticas em português têm vocabulário de domínio bem delimitado (p.ex.: ``gol'', ``voto'', ``inflação'').
        \item \textbf{Embeddings — Limitações no Contexto:}
        \begin{itemize}
          \item \texttt{all-MiniLM-L6-v2} foi treinado majoritariamente em inglês.
          \item \texttt{nomic-embed-text} multimodal: não captura a distinção de classes tão bem em corpus pequeno.
        \end{itemize}
      \end{itemize}
    \end{column}
    \begin{column}{0.43\textwidth}
      \begin{alertblock}{Interpretação}
        Silhueta do TF-IDF é baixa ($\approx 0.02$) indicando que os vetores esparsos são geometricamente superpostos — mas o clustering hierárquico ainda detecta padrões corretos via distâncias de Ward.
      \end{alertblock}
      \vspace{0.3em}
      \begin{block}{DBSCAN/HDBSCAN}
        Inadequados para espaços de alta dimensão — problema da \textit{maldição da dimensionalidade}: distâncias tornam-se indistinguíveis.
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\section{Explicação Semântica via LLM}
% =============================================
% Slide 10: Landmarks e LLM
% =============================================
\begin{frame}{Extração de Landmarks e Explicação por LLM}
  \begin{columns}
    \begin{column}{0.5\textwidth}
      \textbf{\color{ufrnblue}Seleção de Amostras Chave}
      \begin{itemize}
        \item \textbf{Medoides (centrais):} Documentos com menor distância euclidiana ao centroide. Núcleo puro do cluster.
        \item \textbf{Fronteira (ambíguos):} Documentos com menor coeficiente de silhueta individual (próximos a $0$ ou negativos). Indicam interseção temática.
      \end{itemize}
    \end{column}
    \begin{column}{0.5\textwidth}
      \textbf{\color{ufrnblue}Pipeline de Explicação}
      \begin{enumerate}
        \item Selecionar 3 medoides + 3 fronteiristas por cluster
        \item Montar prompt estruturado (1 × por cluster)
        \item Chamar LLM: \textbf{Ollama local} (Qwen 2.5 7B) + \textbf{Gemma 4 31B} (nuvem)
        \item Obter JSON: \texttt{rotulo, resumo, analise\_fronteira, coerencia}
      \end{enumerate}
    \end{column}
  \end{columns}
\end{frame}

% Slide 12: LLM Local vs Nuvem

\begin{frame}{LLM Local (Ollama) vs Nuvem (Gemini/Gemma)}
  \begin{columns}
    \begin{column}{0.5\textwidth}
      \textbf{\color{ufrnblue}Ollama Local — Qwen 2.5 7B}
      \begin{itemize}
        \item \textbf{Privacidade total:} Dados não saem da máquina
        \item \textbf{Sem custo de API:} 100\% gratuito e offline-first
        \item \textbf{Latência:} $\approx$ 1.8s a 2.4s por cluster (GPU local)
        \item \textbf{Fallback automático} para \texttt{llama3.1:8b} em caso de erro
        \item Excelente qualidade gramatical em português
      \end{itemize}
    \end{column}
    \begin{column}{0.5\textwidth}
      \textbf{\color{ufrnblue}Google Gemma 4 31B (API)}
      \begin{itemize}
        \item \textbf{Maior capacidade:} 31B parâmetros vs 7B local
        \item \textbf{Latência:} $\approx$ 0.8s a 1.1s (nuvem)
        \item Retentativas exponenciais contra erros 503 temporários
        \item \textit{Fail-fast} imediato ao atingir cota (429)
        \item Rótulos ligeiramente mais descritivos e detalhados
      \end{itemize}
    \end{column}
  \end{columns}
  \vspace{0.3em}
  \begin{alertblock}{Arquitetura Híbrida}
    O pipeline executa \textbf{em paralelo} ambos os LLMs para cada cluster, permitindo comparação direta da qualidade semântica entre modelos local e em nuvem.
  \end{alertblock}

\section{Conclusão}
% =============================================
% Slide 14: Conclusão
% =============================================
\begin{frame}{Conclusão}
  \begin{itemize}
    \item \textbf{Resultado Principal:} TF-IDF + Agglomerative (Ward) obteve o maior ARI (0.852) neste corpus pequeno de 315 notícias jornalísticas em PT-BR.
    \item \textbf{Embeddings Densos:} Maior qualidade geométrica (Silhueta $\approx 0.06$) e interpretabilidade visual via projeções UMAP/t-SNE — mas ARI inferior ao TF-IDF no corpus testado.
    \item \textbf{DBSCAN/HDBSCAN:} Desempenho limitado em espaços de alta dimensão — maldição da dimensionalidade.
    \item \textbf{Explicação LLM:} Ambos os modelos (Qwen 2.5 local + Gemma 4 31B) produziram rótulos semanticamente coerentes — validando a abordagem híbrida local/nuvem.
    \item \textbf{Pipeline Reproduzível:} Código aberto, configuração via \texttt{.env}, visualizador interativo para análise exploratória.
  \end{itemize}
  \begin{block}{Próximos Passos}
    \small
    Substituir cache JSON por banco vetorial dinâmico (\textbf{Elasticsearch} / \textbf{Qdrant} via Docker) para escalabilidade a centenas de milhares de notícias.
  \end{block}
\end{frame}
